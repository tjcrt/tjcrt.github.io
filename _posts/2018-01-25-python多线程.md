python里面的threading是用来进行多线程编程的。

```python
import threading
import time


class MyThread(threading.Thread):
    def run(self):
        for i in range(3):
            time.sleep(1)
            print("thread {} run , number is {}".format(self.name, i))


def main():
    print("start main thread")
    threads = [MyThread() for i in range(3)]

    for i in threads:
        i.start()

    print("end main thread")

if __name__ == '__main__':
    main()

```
在A线程中调用 B.join() 方法会使得 B 线程阻塞 A 线程（A要等到 B 执行完才执行）
```python
import threading
import time

class MyThread(threading.Thread):
    def run(self):
        for i in range(3):
            time.sleep(1)
            print("thread {} run , number is {}".format(self.name, i))

def main():
    print("start main thread")
    threads = [MyThread() for i in range(3)]

    for i in threads:
        i.start()

    for t in threads:
        t.join()
        time.sleep(1)
    print("end main thread")

if __name__ == '__main__':
    main()
    
```

mutex = threading.Lock() 可以给线程加锁的机制  
mutex.acquire()  
mutex.release()  
```python
import threading
import time

count = 0
mutex = threading.Lock()
class MyThread(threading.Thread):
    def run(self):
        global count
        time.sleep(1)
        if mutex.acquire():
            for i in range(10):
                count += 1
                print("thread {} add 1 , count is {}".format(self.name, count))
                time.sleep(1)
        mutex.release()

def main():
    print("start main thread")
    threads = [MyThread() for i in range(3)]

    for i in threads:
        i.start()

    print("end main thread")

if __name__ == '__main__':
    main()
    
```

mutex = threading.RLock()是可重入锁，一个线程可以获得很多次锁，另一个线程需要等待可重入锁全部释放完才能获得同一个可重入锁。
```python
mutex = threading.RLock()

class MyThread(threading.Thread):

    def run(self):
        if mutex.acquire(1):
            print("thread {} get tmuex".format(self.name))
            time.sleep(1)
            mutex.acquire()
            mutex.release()
            mutex.release()

if __name__ == '__main__':
    print("start main thread...")
    threads = [MyThread() for i in range(3)]
    for t in threads:
        t.start()
    print("end main thread...")

```

con = threading.Condition()是条件锁，用来线程对锁进行条件判断,常用语生产者消费者模型

```python
queue = []

con = threading.Condition()

class MyThread(threading.Thread):

    def run(self):
        if mutex.acquire(1):
            print("thread {} get tmuex".format(self.name))
            time.sleep(1)
            mutex.acquire()
            mutex.release()
            mutex.release()


class Producer(threading.Thread):
    def run(self):
        while True:
            if con.acquire():
                if len(queue) > 100:
                    con.wait()
                else:
                    elem = random.randrange(100)
                    queue.append(elem)
                    print("producer a elem {} , now size is {}".format(elem, len(queue)))
                    time.sleep(random.random())
                    con.notify()
                con.release()

def main():

    for i in range(3):
        Producer().start()

    for i in range(2):
        Consumer().start()

```
